115unsigned int calcChecksum(ifstream* inputFile, bool timeDependent) {
116 if (!inputFile->is_open() || !inputFile->good()) {
117 return 1; // Ошибка открытия файла
118 }
119
120 char ch;
121 unsigned int checksum = 0;
122 unsigned long long index = 0; // Индекс для определения позиции символа
123
124 while (inputFile->get(ch)) {
125
126
127 // Применяем сдвиг и обновляем контрольную сумму, учитывая индекс символа
128 checksum += static_cast<unsigned char>(ch) * (index + 1); // Увеличиваем значение в зависимости от позиции
129 index++; // Увеличиваем индекс
130 }
131
132 if (timeDependent) {
133 // Добавляем текущее время к контрольной сумме
134 checksum += get_minute() % 2; // Пример
135 }
136
137 inputFile->clear();
138 inputFile->seekg(0, ios_base::beg);
139
140 return checksum;
141}
